#include <string.h>

{
    int n, i, j;
    char *word[10], *temp;
    printf("\n Enter how many strings:");
    scanf("%d", &n);
    for (i = 0; i < n; i++)
        word[i] = (char *)malloc(15 * sizeof(char));
    printf("\n Enter %d word:", n);
    for (i = 0; i < n; i++)
        scanf("%s", word[i]);
    for (i = 0; i < n - 1; i++)
    {
        for (j = i + 1; j < n; j++)
        {
            if (strcmp(word[i], word[j]) > 0)
            {
                temp = word[i];
                word[i] = word[j];
                word[j] = temp;
            }
        }
    }
    printf("\n Words in alphabetical order:");
    for (i = 0; i < n; i++)
        printf("\n%s", word[i]);
}

    9. Write a program that accepts n strings and displays the longest string.strings and displays the longest string.¶

    Program¶

    #include <stdio.h>
#include <string.h>
void main()
{
    char s[20][20];
    int i, n;
    int maxlen = 0;
    char longstr[80];
    printf("How many strings you want to enter\n");
    scanf("%d", &n);
    for (i = 1; i <= n; i++)
    {
        printf("Enter %d Strings:", i);
        scanf("%s", &s[i]);
    }
    for (i = 1; i <= n; i++)
    {
        n = strlen(s[i]);
        if (n > maxlen)
        {
            maxlen = n;
            strcpy(longstr, s[i]);
        }
    }
    printf("\nLongest String:= %s", longstr);
}
